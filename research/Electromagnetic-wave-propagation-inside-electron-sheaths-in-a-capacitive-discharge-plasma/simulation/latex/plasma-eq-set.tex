
% 1. Plasma Frequency
\omega_{pe} = \sqrt{ \frac{n_e e^2}{\varepsilon_0 m_e} } ..(1)
% 2. Langmuir Wave Frequency with Thermal component
\omega^2 = \omega_{pe}^2 + 3 k^2 v_{th}^2 ..(2)
% 3. Thermal Velocity (Te already in joules)
v_{th} = \sqrt{ \frac{T_e}{m_e} } ..(3)

% 4. Debye Length
\lambda_D = \sqrt{ \frac{\varepsilon_0 T_e}{n_e e^2} }..(4)

% 5. Plasma Current
I = n e A v_d..(5)
