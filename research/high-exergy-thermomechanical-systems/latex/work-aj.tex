\begin{itemize}
    \item \textbf{Efficiency Definition:}
    \[ \eta = 1 - \frac{Q_{out}}{Q_{in}} \]

    \item \textbf{Heat Transfers:}
    \[ Q_{in} = m C_v (T_2 - T_1) \]
    \[ Q_{out} = m C_v (T_3 - T_4) + m C_p (T_4 - T_1) \]

    \item \textbf{Substitution:}
    \[ \eta = 1 - \frac{m C_v (T_3 - T_4) + m C_p (T_4 - T_1)}{m C_v (T_2 - T_1)} \]
    \[ \eta = 1 - \frac{(T_3 - T_4) + \gamma(T_4 - T_1)}{T_2 - T_1} \]

    \item \textbf{Relating Temperatures to Ratios ($r = \frac{V_3}{V_2}, r_p = \frac{P_2}{P_1}$):}
    \[ T_2 = T_1 r_p \]
    \[ T_3 = T_2 \left(\frac{V_2}{V_3}\right)^{\gamma-1} = T_1 r_p r^{1-\gamma} \]
    \[ T_4 = T_1 \left(\frac{V_4}{V_1}\right) = T_1 r \]
\end{itemize}
