\section*{Efficiency of your custom cycle}

\subsection*{First law}
For a thermodynamic cycle:
\begin{equation}
W_{\text{net}} = Q_{\text{in}} - Q_{\text{out}}
\end{equation}
Thermal efficiency:
\begin{equation}
\eta = \frac{W_{\text{net}}}{Q_{\text{in}}} = 1 - \frac{Q_{\text{out}}}{Q_{\text{in}}}
\end{equation}

\subsection*{Heat in: 1$\rightarrow$2 (isochoric)}
\begin{equation}
Q_{\text{in}} = Q_{12} = m c_v (T_2 - T_1)
\end{equation}

\subsection*{Heat out: 3$\rightarrow$4 (isochoric) + 4$\rightarrow$1 (isobaric)}
\begin{align}
Q_{34} &= m c_v (T_4 - T_3) \Rightarrow |Q_{34}| = m c_v (T_3 - T_4) \\
Q_{41} &= m c_p (T_1 - T_4) \Rightarrow |Q_{41}| = m c_p (T_4 - T_1)
\end{align}
Total heat out:
\begin{equation}
Q_{\text{out}} = m c_v (T_3 - T_4) + m c_p (T_4 - T_1)
\end{equation}
Using $\gamma = c_p/c_v$:
\begin{equation}
Q_{\text{out}} = m c_v \left[ (T_3 - T_4) + \gamma (T_4 - T_1) \right]
\end{equation}

\subsection*{Efficiency formula}
\begin{equation}
\eta = 1 - \frac{Q_{\text{out}}}{Q_{\text{in}}}
= 1 - \frac{m c_v \left[ (T_3 - T_4) + \gamma (T_4 - T_1) \right]}{m c_v (T_2 - T_1)}
\end{equation}
Cancel $m c_v$:
\begin{equation}
\boxed{
\eta = 1 - \frac{T_3 - T_4 + \gamma (T_4 - T_1)}{T_2 - T_1}
}
\end{equation}
