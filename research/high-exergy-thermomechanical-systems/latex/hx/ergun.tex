\[
\frac{\Delta P}{L}
=
\frac{150 (1-\varepsilon)^2 \mu v}{\varepsilon^3 d_p^2}
+
\frac{1.75 (1-\varepsilon) \rho v^2}{\varepsilon^3 d_p}
\]

\text{where:} \\
\Delta P = \text{pressure drop (Pa)} \\
L = \text{bed length (m)} \\
\varepsilon = \text{void fraction (porosity)} \\
\mu = \text{fluid viscosity (Pa·s)} \\
\rho = \text{fluid density (kg/m}^3\text{)} \\
v = \text{superficial velocity (m/s)} \\
d_p = \text{particle diameter (m)}
